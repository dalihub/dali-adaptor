  SET( FILE_DOWNLOAD_PLUGIN_SOURCES ${file_download_plugin_curl_src_files})

  )

    ${DALI_FILE_DOWNLOAD_PLUGIN_CFLAGS}
  )

    ${CURL_INCLUDE_DIRS}
